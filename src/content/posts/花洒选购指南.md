    <p>看准 900-1500 价位的，选择就多了。但我依然优先推荐<strong>主体材质为全铜</strong>，结构上推荐<strong>钢琴键类型</strong>。</p>
    <p>为什么推荐钢琴键？因为我不止一次因为用以前的旋钮款，没看旋钮位置开水湿了衣服。钢琴键的开关逻辑更直观，按压反馈清晰，不容易误操作。</p>
    <p>另外，这个价位可以考虑<strong>恒温功能</strong>。恒温花洒能自动调节冷热水比例，保持出水温度恒定，告别忽冷忽热的困扰。</p>

<script>
  // Water Canvas Animation
  const canvas = document.getElementById('waterCanvas');
  const ctx = canvas.getContext('2d');
  let w, h, particles = [];
  
  function resize() {
    w = canvas.width = canvas.offsetWidth;
    h = canvas.height = canvas.offsetHeight;
  }
  resize();
  window.addEventListener('resize', resize);
  
  class Particle {
    constructor() { this.reset(); }
    reset() {
      this.x = Math.random() * w;
      this.y = Math.random() * h * 0.3 - h * 0.3;
      this.speed = 2 + Math.random() * 4;
      this.size = 1 + Math.random() * 2;
      this.opacity = 0.1 + Math.random() * 0.3;
      this.wobble = Math.random() * Math.PI * 2;
      this.wobbleSpeed = 0.02 + Math.random() * 0.03;
    }
    update() {
      this.y += this.speed;
      this.wobble += this.wobbleSpeed;
      this.x += Math.sin(this.wobble) * 0.5;
      if (this.y > h) this.reset();
    }
    draw() {
      ctx.beginPath();
      ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
      ctx.fillStyle = `rgba(135, 206, 235, ${this.opacity})`;
      ctx.fill();
    }
  }
  
  for (let i = 0; i < 150; i++) particles.push(new Particle());
  
  function animate() {
    ctx.clearRect(0, 0, w, h);
    particles.forEach(p => { p.update(); p.draw(); });
    requestAnimationFrame(animate);
  }
  animate();
  
  // Scroll Reveal
  const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        entry.target.classList.add('visible');
        entry.target.querySelectorAll('.color-item').forEach(item => {
          item.classList.add('visible');
          item.style.setProperty('--fill-width', item.dataset.fill + '%');
        });
      }
    });
  }, { threshold: 0.15 });
  
  document.querySelectorAll('.section').forEach(s => observer.observe(s));
  document.querySelectorAll('.color-item').forEach(item => observer.observe(item));
</script>
